create table users
(
user_id int(3) primary key,
username varchar(20) unique,
pw varchar(20),
num_borrowed int(1) default 0
);

create table books
(
book_id int(3) primary key,
book_name varchar(50),
author varchar(20),
avg_rating double(3, 2) default null,
borrower_id int(3) default null,
FOREIGN KEY (borrower_id) REFERENCES users(user_id)
);

create table borrowings
(
borrower_id varchar(3) not null,
borrowed_id varchar(3) not null,
dob date not null,
dor date not null,
FOREIGN KEY (borrower_id) REFERENCES users(user_id),
FOREIGN KEY (borrowed_id) REFERENCES books(book_id)
);
